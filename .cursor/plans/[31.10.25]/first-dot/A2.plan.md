## UI PAGE STRUCTURE

**Pattern from decision_tree.py**:

```python
# ui/pages/hierarchical.py

def render():
    st.title("🌲 Hierarchical Clustering")
    
    col1, col2 = st.columns([1, 2])
    
    with col1:
        st.subheader("⚙️ Parameters")
        
        # Linkage method
        linkage_method = st.selectbox(
            "Linkage Method",
            ["ward", "single", "complete", "average"],
            help="Ward minimiza varianza, Single usa mínima distancia"
        )
        
        # Distance metric
        metric = st.selectbox(
            "Distance Metric",
            ["euclidean", "manhattan", "cosine"]
        )
        
        # Number of clusters (for cutting)
        n_clusters = st.slider("Number of Clusters", 2, 10, 5)
        
        # Optimal k finder
        st.markdown("---")
        find_optimal = st.checkbox("Find Optimal K (J4)")
        
        if find_optimal:
            k_min = st.number_input("Min K", 2, 10, 2)
            k_max = st.number_input("Max K", 2, 15, 10)
        
        # Train button
        if st.button("🚀 Train", type="primary"):
            # Get data
            X_ready = get_state(StateKeys.X_PREPARED)
            y_ready = get_state(StateKeys.Y_PREPARED)
            
            # Train hierarchical
            linkage_matrix = train_hierarchical(X_ready, linkage_method, metric)
            
            # Cut dendrogram
            labels = cut_dendrogram(linkage_matrix, n_clusters)
            
            # Calculate J4
            j4_score = calculate_j4(X_ready, labels)
            
            # Find optimal k (if requested)
            if find_optimal:
                optimal_k_results = find_optimal_k(
                    linkage_matrix, X_ready, range(k_min, k_max+1)
                )
            
            # Save to state
            set_state("hierarchical_linkage", linkage_matrix)
            set_state("hierarchical_labels", labels)
            
            # Display results...
    
    with col2:
        st.subheader("📊 Results")
        
        # Dendrogram
        st.markdown("#### 🌳 Dendrogram")
        fig_dend = plot_dendrogram(linkage_matrix, ...)
        st.pyplot(fig_dend)
        
        # Cluster distribution
        st.markdown("#### 📊 Cluster Distribution")
        fig_dist = plot_cluster_distribution(labels, y_ready)
        st.pyplot(fig_dist)
        
        # J4 Score
        st.markdown("#### 🎯 J4 Criterion")
        st.metric("J4 Score", f"{j4_score:.3f}")
        st.caption("Higher = mejor separación entre clusters")
        
        # Optimal K chart (if computed)
        if find_optimal:
            st.markdown("#### 📈 Optimal K Finding")
            fig_j4 = plot_j4_curve(optimal_k_results)
            st.pyplot(fig_j4)
            st.info(f"Optimal K: **{optimal_k_results['optimal_k']}** (max J4)")
        
        # Experiment history (inline, like DT)
        st.markdown("---")
        st.markdown("### 📊 Experimentos Guardados")
        # ... (copy pattern from decision_tree.py)
```

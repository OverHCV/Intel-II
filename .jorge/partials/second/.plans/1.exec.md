## IMPLEMENT obj["add_scatter_plots"] {
  
  CREATE IN(@funcs/visualizers.py) OUT(@visualizers.py): {
    function: "plot_interactive_scatter",
    signature: `
      def plot_interactive_scatter(
          X, y, feature_names, 
          x_idx=0, y_idx=1, z_idx=None,
          class_names=None,
          use_plotly=True
      )
    `,
    logic: `
      IF use_plotly AND plotly available:
          IF z_idx is not None:
              Create 3D scatter with plotly.graph_objects
          ELSE:
              Create 2D scatter with plotly.express
          
          Color by y (target class)
          Add hover data with all features
          Interactive zoom, pan, rotate (3D)
      
      ELSE (fallback to matplotlib):
          Create 2D scatter (matplotlib doesn't do 3D well)
          Color by y
          Add legend for classes
          Return matplotlib figure
    `,
    styling: [
      "Plotly: use template='plotly_white'",
      "Matplotlib: use scatter with c=y, cmap='viridis'",
      "Point size: 50 (matplotlib), size=5 (plotly)",
      "Alpha: 0.7 for transparency"
    ]
  }
  
  CREATE IN(@ui/pages/svm/tab.py) OUT(@ui/components/scatter_selector.py): {
    component: "ScatterPlotSelector",
    purpose: "Widget to select axes for scatter plot",
    widgets: [
      "st.selectbox('X Axis', feature_names, index=0)",
      "st.selectbox('Y Axis', feature_names, index=1)",
      "st.checkbox('Enable 3D (Z Axis)', value=False)",
      "IF 3D: st.selectbox('Z Axis', feature_names, index=2)"
    ],
    returns: "(x_idx, y_idx, z_idx or None)"
  }
  
  INTEGRATE IN(@visualizers.py, @scatter_selector.py, @svm/tab.py) OUT(@svm/tab.py): {
    location: "viz_tabs[2] (Data Exploration tab)",
    flow: `
      st.subheader("🗺️ Interactive Feature Scatter")
      st.markdown("Explore relationships between features")
      
      # Axis selectors
      col1, col2, col3 = st.columns(3)
      with col1:
          x_idx = st.selectbox("X Axis", feature_names, index=0)
      with col2:
          y_idx = st.selectbox("Y Axis", feature_names, index=1)
      with col3:
          enable_3d = st.checkbox("Enable 3D", value=False)
      
      z_idx = None
      if enable_3d:
          z_idx = st.selectbox("Z Axis", feature_names, index=2)
      
      # Plot
      fig = plot_interactive_scatter(X, y, feature_names, x_idx, y_idx, z_idx)
      
      IF plotly:
          st.plotly_chart(fig, use_container_width=True)
      ELSE:
          st.pyplot(fig)
    `
